\subsection{Loops}

\begin{ex}
The following code draws a circle.
\begin{lstlisting}
from gymmu.turtle import *

make_turtle()

def draw_circle():
    for i in range(360):
        forward(1)
        turn(1)

draw_circle()
    
show()
\end{lstlisting}

\begin{enumerate}
\item Think about why this draws a circle.
\item can you add a square around this circle?
\end{enumerate}
\end{ex}
